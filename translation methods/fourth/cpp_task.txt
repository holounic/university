;

: AST t[]
| AMP t[]

: RPAR SEMI
| VAR l[]
;

| AMP l[]
;

: RPAR SEMI
| COMMA VAR l[]
;

! COMMA = [','];
! AST = ['\\*'];
! AMP = ['&'];
! LPAR = ['\\('];
! RPAR = ['\\)'];
